def parse_body
  body = gzip? ? gzip_decompress : body_str
  ActiveSupport::JSON.decode(body) if body.present?
end

def total_count
  count = header_str[/X-Total-Count: ([0-9]+)/, 1]
  count.present? ? count.to_i : nil
end
